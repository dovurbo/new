<script>

  const eventList = document.getElementById('event-list');

  const form = document.getElementById('registration-form');

  const formTitle = document.getElementById('form-title');

  const successMessage = document.getElementById('success-message');

  let selectedEvent = '';


  eventList.addEventListener('click', e => {

    if (e.target.tagName === 'LI') {

      selectedEvent = e.target.getAttribute('data-event');

      formTitle.textContent = `Registracija į: ${selectedEvent}`;

      form.style.display = 'block';

      successMessage.style.display = 'none';

      form.name.value = '';

      form.email.value = '';

      form.scrollIntoView({behavior: 'smooth'});

    }

  });


  form.addEventListener('submit', e => {

    e.preventDefault();

    const name = form.name.value.trim();

    const email = form.email.value.trim();


    if (name && email && selectedEvent) {

      // Demo: išvedam duomenis į konsolę

      console.log(`Registracija į "${selectedEvent}" - Vardas: ${name}, El.paštas: ${email}`);


      // Parodom sėkmės žinutę

      form.style.display = 'none';

      successMessage.style.display = 'block';

    }

  });

</script>
